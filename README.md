# React Image Viewer

## Install

- npm

```shell
npm i --save react-image-viewer
```

- yarn

```shell
yarn add react-image-viewer
```

## Usage

```jsx
import React from 'react'

import ImageView from 'react-image-viewer'


class Demo extends React.Compoent {
  render() {
    return (
      <ImageView smallImage={'xxx'} largeImage={'xxx'}/>
    )
  }
}
```
